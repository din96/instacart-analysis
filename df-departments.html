Missing values#

# Search and return mixed Values
print("Mixed data")
for col in df.columns.tolist():
  mixed = (df[[col]].applymap(type) != df[[col]].iloc[0].apply(type)).any(axis = 1)
  if len (df[mixed]) > 0:
    print (col + "True")
  else :
    print(col + ": False")
