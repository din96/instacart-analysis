# Check for mixed values
print("Mixed data")
for col in df.columns.tolist():
  mixed_products = (df[[col]].applymap(type) != df[[col]].iloc[0].apply(type)).any(axis = 1) 
  if len (df[mixed_products]) > 0:
    print (col + ": True")
  else:
