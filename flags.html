def price_label(row):

  if row['prices'] <= 5:
    return 'Low-range product'
  elif (row['prices'] > 5) and (row['prices'] <= 15):
    return 'Mid-range product'
  elif row['prices'] > 15:
    return 'High range'
  else: return 'Not enough data'

  • df.apply(price_label, axis=1).

  • value_counts()

df.loc[df['prices'] <= 5, 'price_range_loc'] = 'Low-range product'
