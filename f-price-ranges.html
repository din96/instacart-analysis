02 Price ranges sorting flag#

# Define sorting function
def price_label(row):

  if row['prices'] <= 5:
    return 'low'
  elif (row['prices'] > 5) and (row['prices'] <= 15):
    return 'mid'
  elif row['prices'] > 15:
    return 'high'
  else: return 'error'

# Apply sorting function to the Dataframe
df['price_range'] = df.apply(price_label, axis=1)
